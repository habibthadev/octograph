# Examples

This page showcases various ways to use Octograph in your React applications. From basic usage to advanced customizations.

## Basic Usage

The simplest way to get started with Octograph:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'

export default function App() {
  return (
    <div className="p-8">
      <h1>My GitHub Activity</h1>
      <OctoGraph username="habibthadev" />
    </div>
  )
}
```
</div>

## With Loading and Error States

Handle different states gracefully:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'

export default function WithStatesExample() {
  return (
    <OctoGraph 
      username="defunkt"
      theme="github"
      loadingComponent={
        <div className="flex items-center justify-center p-8">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-green-500"></div>
          <span className="ml-3">Loading contributions...</span>
        </div>
      }
      errorComponent={({ error, retry }) => (
        <div className="p-4 border border-red-200 rounded bg-red-50">
          <p className="text-red-700">Failed to load: {error.message}</p>
          <button 
            onClick={retry}
            className="mt-2 px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700"
          >
            Try Again
          </button>
        </div>
      )}
    />
  )
}
```
</div>

## Multiple Years Comparison

Show activity across different years:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'
import { useState } from 'react'

export default function YearComparison() {
  const [selectedYear, setSelectedYear] = useState(2023)
  const years = [2021, 2022, 2023]

return (

<div className="space-y-6">
<div className="flex space-x-2">
{years.map(year => (
<button
key={year}
onClick={() => setSelectedYear(year)}
className={`px-4 py-2 rounded ${
              selectedYear === year 
                ? 'bg-blue-500 text-white' 
                : 'bg-gray-200 text-gray-700'
            }`} >
{year}
</button>
))}
</div>

      <OctoGraph
        username="sindresorhus"
        year={selectedYear}
        theme="vercel"
        showStats
        responsive
      />
    </div>

)
}

````
</div>

## Integration with Next.js

Using Octograph in a Next.js application:

interface Props {
username: string
}

export default function ProfilePage({ username }: Props) {
  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 dark:from-gray-900 dark:to-gray-800">
      <div className="container mx-auto px-4 py-8">
        <h1 className="text-3xl font-bold text-center mb-8">
          {username}'s GitHub Activity
        </h1>
        
        <div className="max-w-4xl mx-auto">
          <OctoGraph 
            username={username}
            theme="supabase"
            showStats
            responsive
            cellRadius={2}
          />
        </div>
      </div>
    </div>
  )
}

export const getServerSideProps: GetServerSideProps = async ({ params }) => {
  return {
    props: {
      username: params?.username || 'habibthadev',
    },
  }
}

## Dashboard Example

Building a complete GitHub activity dashboard:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'
import { useState } from 'react'

const users = [
{ username: 'habibthadev', name: 'The habibthadev' },
{ username: 'torvalds', name: 'Linus Torvalds' },
{ username: 'gaearon', name: 'Dan Abramov' },
]

export default function Dashboard() {
  const [selectedTheme, setSelectedTheme] = useState('github')
  const themes = ['github', 'shadcn', 'vesper', 'vercel']

return (

<div className="min-h-screen bg-gray-50 dark:bg-gray-900 p-8">
<div className="max-w-7xl mx-auto">
<header className="mb-8">
<h1 className="text-4xl font-bold text-gray-900 dark:text-white mb-4">
GitHub Activity Dashboard
</h1>

          <div className="flex items-center space-x-4">
            <label className="text-sm font-medium text-gray-700 dark:text-gray-300">
              Theme:
            </label>
            <select
              value={selectedTheme}
              onChange={(e) => setSelectedTheme(e.target.value)}
              className="border border-gray-300 rounded px-3 py-1 dark:bg-gray-800 dark:border-gray-600"
            >
              {themes.map(theme => (
                <option key={theme} value={theme}>
                  {theme.charAt(0).toUpperCase() + theme.slice(1)}
                </option>
              ))}
            </select>
          </div>
        </header>

        <div className="grid grid-cols-1 lg:grid-cols-2 xl:grid-cols-3 gap-8">
          {users.map(user => (
            <div key={user.username} className="bg-white dark:bg-gray-800 rounded-lg p-6 shadow-lg">
              <h2 className="text-xl font-semibold mb-4 text-gray-900 dark:text-white">
                {user.name}
              </h2>
              <OctoGraph
                username={user.username}
                theme={selectedTheme}
                responsive
                showStats
                cellSize={10}
              />
            </div>
          ))}
        </div>
      </div>
    </div>

)
}

````
</div>

## Real-time Updates

Keep contribution data fresh with periodic updates:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'
import { useEffect, useState } from 'react'

export default function RealtimeExample() {
  const [lastUpdated, setLastUpdated] = useState<Date>(new Date())

  useEffect(() => {
    const interval = setInterval(() => {
      setLastUpdated(new Date())
    }, 300000) // Update every 5 minutes

    return () => clearInterval(interval)
  }, [])

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-bold">Live Activity</h2>
        <div className="text-sm text-gray-500">
          Last updated: {lastUpdated.toLocaleTimeString()}
        </div>
      </div>

      <OctoGraph
        username="microsoft"
        theme="ayu"
        key={lastUpdated.getTime()} // Force re-render
        showStats
        cacheDuration="5m"
      />
    </div>
  )
}
````

</div>

## Performance Optimized

For applications with many graphs:

<div className="mt-4 p-4 border rounded-lg bg-gray-50 dark:bg-gray-900">
```tsx
import { OctoGraph } from 'octograph'
import { memo, useMemo } from 'react'

const OptimizedGraph = memo(({ username, theme }: { username: string, theme: string }) => {
return (

<OctoGraph
  username={username}
  theme={theme}
  lazy // Only load when in viewport
  cacheDuration="1h"
  mode="lightweight"
/>
) })

export default function PerformanceExample() {
  const users = useMemo(() => [
    'habibthadev', 'torvalds', 'gaearon', 'sindresorhus', 'kentcdodds'
  ], [])

return (

<div className="space-y-8">
  {users.map((username) => (
    <div key={username} className="border-b pb-8">
      <h3 className="text-lg font-semibold mb-4">{username}</h3>
      <OptimizedGraph username={username} theme="atom" />
    </div>
  ))}
</div>
) }

```
</div>
